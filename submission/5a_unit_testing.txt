from EmotionDetection.emotion_detection import emotion_detection
import unittest

class TestEmotionDetection(unittest.TestCase):
    def test_emotion_detection(self):

        # Test Case 1 - Joy
        response_1 = emotion_detection("I am glad this happened")
        self.assertEqual(response_1["dominant_emotion"], "joy")

        # Test Case 2 - Anger
        response_2 = emotion_detection("I am really mad about this")
        self.assertEqual(response_2["dominant_emotion"], "anger")

        # Test Case 3 - Disgust
        response_3 = emotion_detection("I feel disgusted just hearing about this")
        self.assertEqual(response_3["dominant_emotion"], "disgust")

        # Test Case 4 - Sadness
        response_4 = emotion_detection("I am so sad about this")
        self.assertEqual(response_4["dominant_emotion"], "sadness")

        # Test Case 5 - Feat
        response_5 = emotion_detection("I am really afraid that this will happen")
        self.assertEqual(response_5["dominant_emotion"], "fear")


unittest.main()
